3.SELECT VYKDYTOJAI.VARDAS, VYKDYTOJAI.PAVARDE, PROJEKTAI.PAVADINIMAS FROM VYKDYTOJAI JOIN VYKDYMAS ON VYKDYTOJAI.NR = VYKDYMAS.VYKDYTOJAS JOIN PROJEKTAI ON VYKDYMAS.PROJEKTAS = PROJEKTAI.NR WHERE VYKDYMAS.PROJEKTAS = 2

5.SELECT VYKDYTOJAI.VARDAS, VYKDYTOJAI.PAVARDE, PROJEKTAI.PAVADINIMAS FROM VYKDYTOJAI JOIN VYKDYMAS ON VYKDYTOJAI.NR = VYKDYMAS.VYKDYTOJAS JOIN PROJEKTAI ON VYKDYMAS.PROJEKTAS = PROJEKTAI.NR WHERE VYKDYTOJAI.PAVARDE LIKE '%Ä—' AND VYKDYMAS.STATUSAS = 'Vadovas';
6.SELECT PROJEKTAI.PAVADINIMAS, COUNT(PROJEKTAI.PAVADINIMAS) FROM VYKDYTOJAI JOIN VYKDYMAS ON VYKDYTOJAI.NR = VYKDYMAS.VYKDYTOJAS JOIN PROJEKTAI ON VYKDYMAS.PROJEKTAS = PROJEKTAI.NR GROUP BY PROJEKTAI.PAVADINIMAS;
7.SELECT PROJEKTAI.PAVADINIMAS, COUNT(PROJEKTAI.PAVADINIMAS) FROM VYKDYTOJAI JOIN VYKDYMAS ON VYKDYTOJAI.NR = VYKDYMAS.VYKDYTOJAS JOIN PROJEKTAI ON VYKDYMAS.PROJEKTAS = PROJEKTAI.NR GROUP BY PROJEKTAI.PAVADINIMAS HAVING COUNT(PROJEKTAI.PAVADINIMAS) < 4;
8.SELECT VYKDYTOJAI.VARDAS, VYKDYTOJAI.PAVARDE, PROJEKTAI.PAVADINIMAS FROM VYKDYTOJAI JOIN VYKDYMAS ON VYKDYTOJAI.NR = VYKDYMAS.VYKDYTOJAS JOIN PROJEKTAI ON VYKDYMAS.PROJEKTAS = PROJEKTAI.NR WHERE NOT VYKDYMAS.STATUSAS = 'Vadovas';
